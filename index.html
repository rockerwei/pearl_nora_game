const CACHE_NAME = "kids-game-cache-v6"; const urlsToCache = [ "./", "./index.html", "./manifest.json" ]; self.addEventListener("install", event => { self.skipWaiting(); event.waitUntil( caches.open(CACHE_NAME).then(cache => cache.addAll(urlsToCache)) ); }); self.addEventListener("activate", event => { event.waitUntil( caches.keys().then(keys => Promise.all( keys.filter(key => key !== CACHE_NAME).map(key => caches.delete(key)) ) ) ); self.clients.claim(); });
